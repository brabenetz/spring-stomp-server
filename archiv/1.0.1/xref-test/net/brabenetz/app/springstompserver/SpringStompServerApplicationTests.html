64      public void cleanupSession() {
65          this.subscriptions.forEach(consumer -> consumer.unsubscribe());
66          if (this.currentSession != null) {
67              this.currentSession.disconnect();
68          }

89      private void testStompWebsocketEndpoint(String clientName, String websocketEndpoint, WebSocketStompClient client)
90              throws Exception {
91          // connect to the Websocket
92          WebSocketStompSessionHandlermpSessionHandler.html#WebSocketStompSessionHandler">WebSocketStompSessionHandler sessionHandler = new WebSocketStompSessionHandler(clientName);
93          ListenableFuture<StompSession> stompClientConnection = client.connect(websocketEndpoint, sessionHandler);
94          // wait until Session is created
95          this.currentSession = stompClientConnection.get(1, TimeUnit.SECONDS);
96  
97          // container to collect and sync the result
98          CountDownLatch doneSignal = new CountDownLatch(1);
99          List<WebSocketPayload<String>> messages = new ArrayList<>();
100 
101         // Register Subscription
102         subscriptions.add(this.currentSession.subscribe("/topic/test/1234", WebSocketUtils.createStompFrameHandler((StompHeaders headers, String payload) -> {
103             LOG.info(clientName + " FrameHandler got new Payload: " + payload);
104             messages.add(new WebSocketPayload<>(headers, payload));
105             doneSignal.countDown();
106             headers.forEach((k, v) -> {
107                 LOG.info(String.format(clientName + " HEADER: %s = %s", k, v));
108                 // Example Output:
109                 // STOMP-WebSocket HEADER: destination = [/topic/test/1234]
110                 // STOMP-WebSocket HEADER: content-type = [application/json]
111                 // STOMP-WebSocket HEADER: subscription = [0]
112                 // STOMP-WebSocket HEADER: message-id = [6ff3affb-eacd-ea53-b403-60dfdf8fcd88-0]
113                 // STOMP-WebSocket HEADER: content-length = [6]
114             });
115         })));
116 
117         // Send Message
118         this.currentSession.send("/topic/test/1234", "test");
119 
120         // Wait until Message is consumed
121         doneSignal.await(1, TimeUnit.SECONDS);
122         assertThat(doneSignal.getCount()).describedAs("CountDownLatch").isEqualTo(0);
123 
124         // Verify that consumed Message Payload.
125         assertThat(messages).hasSize(1);
126         assertThat(messages.get(0).getBody()).isEqualTo("test");
127         assertThat(messages.get(0).getHeaders().get("destination")).containsExactly("/topic/test/1234");
128         assertThat(messages.get(0).getHeaders().get("content-type")).containsExactly("application/json");
129     }
